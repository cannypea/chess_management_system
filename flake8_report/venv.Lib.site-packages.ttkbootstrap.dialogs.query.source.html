153 """Validate the data before closing."""
154 # no default checks required for string data types,
155 # unless there is a list of items to pick from
156 if self._datatype not in [float, int, complex] and (self._items is None or len(self._items) == 0):
157 return True

166 title=MessageCatalog.translate("Invalid data type"),
167 parent=self._toplevel,
168 )
169 return False

173 msg = MessageCatalog.translate("Number cannot be greater than")

179 return False

183 msg = MessageCatalog.translate("Number cannot be less than")
