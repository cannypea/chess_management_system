1 r"""Validation framework for ttkbootstrap entry widgets.
2  
3 This module provides classes and functions for adding validation to Entry,
4 Spinbox, and Combobox widgets. When validation fails, a 'danger' colored
5 border is applied to the widget, which disappears when the contents become valid.
6  
7 The module includes:
8 - Predefined validation functions (text, numeric, phone number, regex, etc.)
9 - Custom validation decorator (@validator)
10 - Helper functions starting with "add_" prefix for quick validation setup
11  
12 Classes:
13 ValidationEvent: Contains attributes of a validation event from tkinter
14  
15 Functions:
16 add_validation: Core function to add validation to any compatible widget
17 add_text_validation: Validate that contents is alphabetic text
18 add_numeric_validation: Validate that contents is numeric
19 add_phonenumber_validation: Validate phone number format
20 add_regex_validation: Validate against custom regex pattern
21 add_range_validation: Validate numeric value is within range
22 add_option_validation: Validate value is in list of options
23  
24 Example:
25 Using predefined validation:
26 ```python
27 import ttkbootstrap as ttk
28 from ttkbootstrap.validation import *
29  
30 app = ttk.Window()
31 entry = ttk.Entry()
32 entry.pack(padx=10, pady=10)
33  
34 # Check if contents is text
35 add_text_validation(entry)
36  
37 # Prevent any entry except text
38 add_text_validation(entry, when='key')
39  
40 # Check for specific list of options
41 add_option_validation(entry, ['red', 'blue', 'green'])
42  
43 # Validate against regex expression
44 add_regex_validation(entry, r'\d{4}-\d{2}-\d{2}')
45  
46 app.mainloop()
47 ```
48  
49 Creating custom validation:
50 ```python
51 from ttkbootstrap.validation import validator, add_validation
52  
53 @validator
54 def validate_long_text(event):
55 if len(event.postchangetext) > 20:
56 return True
57 else:
58 return False
59  
60 # Apply custom validation
61 add_validation(entry, validate_long_text)
62 ```

71 class ValidationEvent:
72 """Contains the attributes of a validation event returned by the
73 `validatecommand` on a tkinter widget.
74  
75 Attributes:
76  
77 actioncode (str):
78 0 for an attempted deletion, 1 for an attempted insertion,
79 or -1 if the callback was for focusin, focusout, or a
80 change to the textvariable.
81  
82 insertdeleteindex (str):
83 When the user attempts to insert or delete text, this
84 attribute will be the index of the beginning of the
85 insertion or deletion. If the callback was due to focusin,
86 focusout, or a change to the textvariable, the attribute
87 will be -1.
88  
89 postchangetext (str):
90 The value that the text will have if the change is allowed.
91  
92 prechangetext (str):
93 The text in the entry before the change.
94  
95 insertdeletetext (str):
96 The text inserted or deleted if the call was due to an
97 insertion or deletion.
98  
99 validationtype (str):
100 Specifies the widget's validation option which specifies
101 _when_ the validation will occur.
102  
103 validationreason (str):
104 The reason for the validation callback (key, focusin,
105 focusout, forced).
106  
107 widget (Widget):
108 The widget object that is being validated.

143 def add_validation(widget: Misc, func: Callable[..., bool], when: str = "focusout", **kwargs: Any) -> None:
144 """Adds validation to the widget of type `Entry`, `Combobox`, or
145 `Spinbox`. The func should accept a parameter of type
146 `ValidationEvent` and should return a boolean value.
147  
148 Parameters:
149  
150 widget (Widget):
151 The widget on which validation will be applied.
152  
153 func (Callable):
154 The function that will be called when a validation event
155 occurs.
156  
157 when (str):
158 Indicates when the validation event should occur. Possible
159 values include:
160  
161 * focus - whenever the widget gets or loses focus
162 * focusin - whenever the widget gets focus
163 * focusout - whenever the widget loses focus
164 * key - whenever a key is pressed
165 * all - validate in all of the above situations
166  
167 kwargs (Dict):
168 Optional arguments passed to the callback.
169 """
170 f = widget.register(lambda *e: func(*e, **kwargs))
171 subs = (r"%d", r"%i", r"%P", r"%s", r"%S", r"%v", r"%V", r"%W")
172 widget.configure(validate=when, validatecommand=(f, *subs))

176 def _validate_text(event: ValidationEvent) -> bool:
177 """Contents is text."""

179 return True
