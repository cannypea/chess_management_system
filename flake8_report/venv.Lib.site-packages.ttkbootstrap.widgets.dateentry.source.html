1 """DateEntry widget for ttkbootstrap.

3 This module provides the DateEntry widget, which combines an Entry field

39 """A date entry widget combines an Entry field and a Button for date selection.
40  
41 When the button is pressed, a calendar popup is displayed allowing the user
42 to select a date. The selected date is inserted into the entry field.
43  
44 The <<DateEntrySelected>> event is generated when a date is selected from
45 the calendar popup.
46  
47 Features:
48 - Configurable date format using strftime format strings
49 - Customizable starting weekday (0=Monday, 6=Sunday)
50 - Style customization via bootstyle parameter
51 - Date validation with optional exception raising
52 - Access to entry and button widgets via instance attributes
53  
54 The date chooser popup will use the date in the entry field as the initial
55 focus date if it matches the specified dateformat. By default, the format
56 is locale-specific ("%x").
57  
58 The bootstyle parameter can be used to change the widget colors. Available
59 options include: primary, secondary, success, info, warning, danger, light, dark.
60  
61 Widget Attributes:
62 entry (ttk.Entry): The entry field displaying the selected date
63 button (ttk.Button): The button that opens the calendar popup
64  
65 ![](../../assets/widgets/date-entry.png)
66 """

252 def _validate_dateformat(dateformat: str) -> str:
253 """Validate that a date format string is appropriate for dates.
254  
255 Checks that the format string contains sufficient information to
256 represent a complete date (year, month, and day).
257  
258 Parameters:
259 dateformat (str): The strftime format string to validate.
260  
261 Returns:
262 str: The validated format string.
263  
264 Raises:
265 ValueError: If the format string cannot be used to represent
266 a complete date.
267  
268 See Also:
269 https://docs.python.org/3/library/datetime.html#strftime-and-strptime-behavior
270 """
271 has_year: bool = any(y in dateformat for y in ('%Y', '%y', '%G'))
272 has_month: bool = any(m in dateformat for m in ('%m', '%B', '%b'))
273 has_day: bool = any(d in dateformat for d in ('%d',))
274 is_full_format: bool = any(f in dateformat for f in ('%x', '%c'))
275  
276 if has_year and has_month and has_day:
277 return dateformat
278  
279 if is_full_format:
280 return dateformat
281  
282 # Special case: (day of the year & year)
283 if '%j' in dateformat and has_year:
284 return dateformat
285  
286 # Special case: (week day & week number & year)
287 has_week_number: bool = any(w in dateformat for w in ('%U', '%W', '%V'))
288 has_week_day: bool = any(w in dateformat for w in ('%a', '%A', '%w'))
289 if has_week_number and has_week_day and has_year:
290 return dateformat
291

  • E501 Line too long (156 > 119 characters)
293 f'Given formatting string ("{dateformat}"), cannot be used to validate a given strings for dates or display a given datetime object as a date!')

297 """Strip time components from a datetime object.

314 """Set the currently selected date.

354 def _on_date_ask(self) -> None:
355 """Handle the calendar button click event.
356  
357 Opens the date selection popup and updates the entry field with the
358 selected date. Generates the <<DateEntrySelected>> event when a date
359 is chosen.
360  
361 Raises:
362 ValueError: If raise_exception is True and the entry text doesn't
363 match the configured date format.
364 """
365 from warnings import warn
366  
367 currently_selected_date: str = self.entry.get() or datetime.today().strftime(self.__dateformat)
368 try:
369 self._startdate: datetime = datetime.strptime(currently_selected_date, self.__dateformat)
370 except ValueError as exc:

372 if self._raise_exception:

378 new_date = Querybox.get_date(
